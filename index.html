-- 新增 products 資料表
CREATE TABLE products (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  name text NOT NULL,
  code text,
  category text,
  unit text,
  stock numeric DEFAULT 0,
  min_stock numeric DEFAULT 0,
  cost numeric,
  price numeric,
  notes text,
  created_at timestamptz DEFAULT now()
);

-- 新增 customers 資料表
CREATE TABLE customers (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  name text NOT NULL,
  contact text,
  phone text,
  email text,
  address text,
  type text DEFAULT '批發',
  credit_limit numeric,
  notes text,
  created_at timestamptz DEFAULT now()
);

-- entries 新增 單價 欄位（若原本沒有）
ALTER TABLE entries ADD COLUMN IF NOT EXISTS 單價 numeric DEFAULT 0;

-- 到貨回報資料表
CREATE TABLE IF NOT EXISTS deliveries (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  code text NOT NULL,
  supplier text,
  description text,
  expected_count numeric,
  eta date,
  office_notes text,
  status text DEFAULT 'pending',
  arrived_at timestamptz,
  actual_count numeric,
  field_notes text,
  reporter text,
  created_by text,
  created_at timestamptz DEFAULT now()
);

-- 刪除記錄資料表
CREATE TABLE IF NOT EXISTS delete_logs (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  table_name text NOT NULL,
  record_id text,
  record_summary text,
  deleted_by text,
  deleted_at timestamptz DEFAULT now()
);
